CREATE TABLE `h_login` (
  `SYSCODE` varchar(32) NOT NULL,
  `LOGINIP` varchar(20) DEFAULT NULL,
  `LOGINTIME` varchar(20) DEFAULT NULL,
  `USERCODE` varchar(32) DEFAULT NULL,
  `USERNAME` varchar(40) DEFAULT NULL,
  `UNITSYS` varchar(32) DEFAULT NULL,
  `FLAG` varchar(40) DEFAULT NULL,
  `REMARK` varchar(40) DEFAULT NULL,
  PRIMARY KEY (`SYSCODE`),
  KEY `IDX_L_LOGIN_UNITSYS` (`UNITSYS`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=gbk COMMENT='用户登录日志信息表';
